I - Alphabetical syntax summary

I.1 Overview

This document describes the syntax of the Eiffel language using a meta language and is arranged in alphabetical order.

A structured syntax description arranged in top-down order is also available.

I.2

Actual ::=
Expression | Address
Actual_generics ::=
"[" Type_list "]"
Actual_list ::=
{Actual "," ...}
Actuals ::=
"(" Actual_list ")"
Address ::=
"$" Address_mark
Address_mark ::=
Feature_name | Current | Result
Anchor ::=
Identifier | Current
Anchored ::=
like Anchor
Assertion ::=
{Assertion_clause Separator ...}
Assertion_clause ::=
[Tag_mark] Unlabeled_assertion_clause
Assignment ::=
Writable ":=" Expression
Assignment_attempt ::=
Writable "?=" Expression
Attribute ::=
Identifier
Attribute_list ::=
{ Identifier ","
Binary ::=
"+" | "-" | "*" | "/" |
"<" | ">" | "<=" |
"//" | "\\" | "^" |
and | or | xor |
and then | or else | implies
Binary_expression ::=
Expression Infix_operator
Bit_constant ::=
Bit_sequence
Bit_type ::=
BIT Constant
Boolean_constant ::=
true | false
Boolean_expression ::=
Expression
Call ::=
[Parenthesized_qualifier] Call_chain
Call_chain ::=
{Unqualified_call "." ...}+
Check ::=
check Assertion end
Choice ::=
Constant | Interval
Choices ::=
{Choice "," ...}
Class_declaration ::=

Class_header

[Obsolete]
[Inheritance]
[Creators]
[Features]
[Invariant]
end

[Header_mark] class Class_name

{Class_name "," ...}
Class_name ::=

Class_name [Actual_generics]
Class_type_expanded ::=

Clients ::=

Comparison ::=
"=" | "/="

{Instruction Separator ...}

Constant ::=

Entity

Constraint ::=
"->" Restricted_type
Creation ::=
"!" [Type] "!" Writable [Creation_call]
Creation_call ::=
"." Unqualified_call
Creation_clause ::=
[Clients][Header_comment] Feature_list
Creators ::=
creation {Creation_clause creation ...}+
Debug ::=
debug [Debug_keys] Compound end
Debug_key ::=
Manifest_string
Debug_key_list ::=
{Debug_key "," ...}
Debug_keys ::=

Declaration_body ::=
[Formal_arguments]
[Type_mark]
[Constant_or_routine]
Deferred ::=
deferred
Effective ::=
Internal | External
Else_part ::=
else Compound
Entity ::=
Writable | Read_only
Entity_declaration_group ::=
Identifier_list Type_mark
Entity_declaration_list ::=
{Entity_declaration_group Separator ...}
Equality ::=

Exit ::=

Expression ::=
Call | Operator_expression
Manifest_constant | Manifest_array
Old | Strip
Expression_list ::-
{ Expression "," ...}

[Rename]
[New_exports]
[Undefine]
[Redefine]
[Select]
end

[Clients]
[Header_comment]
Feature_declaration_list

Formal_generic_name ::=

Indexing ::=

infix Infix_name

Binary | Free_operator
Inheritance ::=

from Compound

Creation |

Assignment_attempt |

Internal ::=
Routine_mark Compound

Invariant ::=
invariant Assertion

Manifest_string

end

Exit loop Compound

Manifest_string

New_feature_list ::=

obsolete Message
Old ::=
old Expression
Operator_expression ::=

Parent ::=

Parenthesized_qualifier ::=
Parenthesized "."
Postcondition ::=

Precondition ::=
require [else] Assertion
Prefix ::=

Prefix_name ::=
Manifest_string
Prefix_operator ::=

Read_only ::=
Formal | Current

redefine Feature_list
Rename ::=
